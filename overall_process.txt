14.start_capturing(AppState *app)
	enum v4l2_buf_type type;
	for(struct v4l2_buffer buf)	xioctl(app->fd,VIDIOC_QBUF,&buf)
	//注意struct v4l2_buffer 是通信协议，app->buffers 是本地记录
	//两者职责不同，VIDIOC_QBUF 只接受前者。就像快递单号和包裹本身，快递员认单号不认包裹长什么样。
	//所以不能使用app->buffers来VIDIOC_QBUF
	// 注意init_mmap 中的 buf 和 start_capturing 中的 buf是两个完全独立的局部变量，没有任何关系
	// 驱动只看 index 字段
	// index 相同 = 操作的是同一个内核缓冲区
	//就像酒店房卡，不管你拿的是哪把钥匙，只要房间号是302，开的就是302的门。
	type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
	xioctl(app->fd,VIDIOC_STREAMON,&type)//驱动开始让摄像头往缓冲区写数据
	
	
	
	
15.capture_start_thread(AppState *app)
	SDL_CreateThread(capture_thread,"capture_thread",app);
		|
	capture_thread(void *userdate)
		|
	fd_set fds;
        struct timeval tv;
        select(app->fd+1,&fds,NULL,NULL,&tv)
        		select				cond
        	等待对象	文件描述符（硬件/内核）		软件条件
        	谁来唤醒	内核自动通知		另一个线程主动 Signal
        	超时		✅ 原生支持		✅ CondWaitTimeout
        	使用场景	IO就绪、网络、设备		线程间同步
        	需要mutex		❌		✅ 必须配合mutex
        	一句话：select 是等硬件，cond 是等同伴。
        	|
        struct v4l2_buffer buf;
        xioctl(app->fd,VIDIOC_DQBUF,&buf)
        	|
        yuyv_to_rgb24(app->buffers[buf.index].start, ...) // 源：mmap共享内存    目标：线程私有缓冲
        buf（用户空间结构体）
	  ├─ index/bytesused/timestamp 等元数据  → 直接用 ✅
	  └─ m.offset（指向内核缓冲区）          → 不能直接用 ❌
		                                      ↓ mmap 搭桥
		               app->buffers[index].start ✅
        SDL_LockMutex(app->latest.mutex);
        memcpy(app->latest.rgb, app->capture_rgb, app->capture_rgb_bytes);
        SDL_UnlockMutex(app->latest.mutex);
        stream_push_rgb_frame(app,app->capture_rgb);	record_push_rgb_frame(app,app->capture_rgb)
	xioctl(app->fd,VIDIOC_QBUF,&buf)
